Schema

Client(
	id,
	age,
	fname,
	sname,
	guardian REFERENCES Guardian.id,
	fellowship REFERENCES Fellowship.id
);

Guardian(
	phoneNumber,
	fname,
	sname
);

Fellowship(
	id,
	name,
	creator REFERENCES Guardian.id
);

Journey(
	id,	
	fellowship REFERENCES Group.id,
	supervisor REFERENCES Guardian.id,
	startTime,
	endTime,
	startLatitude,
	startLongitude,
	endLatitude,
	endLongitude
);

ChildGroupAuth(
	child REFERENCES Client.id,
	guardian REFERENCES Guardian.id
);
